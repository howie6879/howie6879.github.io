вҖҰ… вҖҰ…

йӮЈжҺҘдёӢжқҘйңҖиҰҒи®Ёи®әзҡ„пјҢе°ұжҳҜ src зҡ„еҶ…йғЁз»“жһ„иҜҘжҳҜд»Җд№Ҳж ·зҡ„е‘ўпјҹ

жң¬з« е°ҶеҶҷдёҖдёӘ rss и§Јжһҗеұ•зӨәзҡ„йЎ№зӣ®з”ЁеҒҡжј”зӨәгҖӮ

2.1.жҷ®йҖҡзҡ„йЎ№зӣ®з»“жһ„

дёҖдёӘжҷ®йҖҡзҡ„йЎ№зӣ®пјҡ

  • дёҚйңҖиҰҒж·»еҠ еҗҺз»ӯжЁЎеқ—еҠҹиғҪ
  • еҝ«йҖҹејҖеҸ‘дҪҝз”ЁпјҢдёҚйңҖиҰҒз»ҙжҠӨ
  • ж— иҫғеӨҚжқӮзҡ„еүҚз«ҜйңҖжұӮ
  • з”Ёе®Ңе°ұиө°

йӮЈд№Ҳе°ұеҸҜд»ҘеғҸ demo01 дёӯдёҖж ·пјҢеҸӘйңҖиҰҒж·»еҠ дёҖдёӘ run.py жҲ–иҖ…еҸ«еҒҡ app.py ж–Үд»¶пјҲеҸҚжӯЈиҝҷжҳҜдёҖдёӘеҗҜеҠЁж–Үд»¶пјҢе‘ҪеҗҚеҸҜйҡҸж„ҸпјүпјҢдёҚи®әжҳҜй…ҚзҪ®гҖҒи·Ҝз”ұйғҪеҶҷеңЁдёҖиө·е°ұеҘҪдәҶгҖӮ

ж–°е»әдёҖдёӘйЎ№зӣ®еҰӮдёӢпјҡ

д»»ж„ҸдёҖдёӘ rss жәҗпјҢеҒҮи®ҫйЎ№зӣ®йңҖиҰҒе°Ҷе…¶дёӯзҡ„ж–Үз« ж Үйўҳд»ҘеҸҠй“ҫжҺҘжҸҗеҸ–е№¶еұ•зӨәеҮәжқҘпјҢжҜ”еҰӮд»Ҙjsonж јејҸиҝ”еӣһпјҢиҝҷеұһдәҺеҫҲз®ҖеҚ•зҡ„еҠҹиғҪпјҢеҸҜд»ҘиҜҙеҸӘжңүдёҖж®өйҖ»иҫ‘пјҢrun.py еҶ…е®№еҰӮдёӢпјҡ

#!/usr/bin/env python
from sanic import Sanic
from sanic.response import json
from feedparser import parse

app = Sanic()


@app.route("/")
async def index(request):
    url = "http://blog.howie6879.cn/atom.xml"
    feed = parse(url)
    articles = feed['entries']
    data = []
    for article in articles:
        data.append({"title": article["title_detail"]["value"], "link": article["link"]})
    return json(data)


if __name__ == "__main__":
    app.run(host="0.0.0.0", port=8000)

и®ҝй—® http://0.0.0.0:8000/пјҢдјҡиҝ”еӣһдёҖдёІjsonпјҢеҰӮдёӢпјҡ

жҺҘдёӢжқҘпјҢй—®йўҳеҚҮзә§пјҢжҲ‘жғіиҰҒе°Ҷж Үйўҳй“ҫжҺҘз”ЁйЎөйқўеұ•зӨәпјҢиҜҘжҖҺд№Ҳеј„пјҹ

жҲ‘д»¬йңҖиҰҒдёҖдёӘйЎөйқўжЁЎжқҝпјҢз„¶еҗҺеҶҷе…Ҙjsonж•°жҚ®пјҢжңҖеҗҺз”Ё jinja2 зҡ„ template е°Ҷе…¶жёІжҹ“пјҡ

# templateзҡ„жёІжҹ“жЁЎжқҝ
# еҗҺйқўдјҡдҪҝз”Ёжӣҙж–№дҫҝзҡ„жЁЎжқҝеј•з”Ёж–№ејҸ

# йҖ»иҫ‘еҮҪж•°
@app.route("/html")

    feed = parse(url)
    articles = feed['entries']

    html_content = template.render(articles=data)

е…·дҪ“з»“жһ„д»Јз Ғи§Ғsample01пјҢиҝҗиЎҢиө·жқҘпјҢз„¶еҗҺиҫ“е…Ҙ http://0.0.0.0:8000/html е°ұеҸҜд»ҘзңӢеҲ°иў«еұ•зӨәеҮәжқҘзҡ„йЎөйқў^_^

еҒҮи®ҫйңҖиҰҒзј–еҶҷеүҚз«ҜйЎөйқўжҜ”иҫғеӨҡпјҢйӮЈд№ҲдҪ е°ұйңҖиҰҒж·»еҠ static д»ҘеҸҠ templates ж–Үд»¶еӨ№з”ЁжқҘз®ЎзҗҶеҗ„дёӘз•ҢйқўжЁЎеқ—пјҢе…·дҪ“дёӢйқўдјҡд»Ӣз»ҚгҖӮ

2.2.йЎ№зӣ®з»“жһ„е…·дҪ“иҜҙжҳҺ

еҪ“зј–еҶҷзҡ„йЎ№зӣ®иҝҮдәҺеӨҚжқӮпјҢжҲ‘йғҪдјҡе°Ҷе…¶еҪ“еҒҡдёҖдёӘз¬¬дёүж–№еҢ…жқҘз®ЎзҗҶйЎ№зӣ®дёӯж¶үеҸҠзҡ„еҗ„з§ҚжЁЎеқ—пјҢжҜ”еҰӮ owllookпјҢзӣ®еҪ•дёӢйқўдҪ дјҡеҸ‘зҺ°жңүдёӘ __init__.py ж–Үд»¶пјҢе®ғеҲқе§ӢеҢ–дәҶеҪ“еүҚзӣ®еҪ•дёӢзҡ„еә”з”ЁпјҢз„¶еҗҺд»Јз Ғдёӯеј•з”ЁжҹҗдёӘеҮҪж•°еҸҜд»Ҙиҝҷд№ҲеҶҷпјҡ

from owllook.config import Config

enable_async = sys.version_info >= (3, 6)

app = Sanic()

# jinjia2 config
env = Environment(
    loader=PackageLoader('views.rss', '../templates'),
    autoescape=select_autoescape(['html', 'xml', 'tpl']),
    enable_async=enable_async)


async def template(tpl, **kwargs):
    template = env.get_template(tpl)
    rendered_template = await template.render_async(**kwargs)
    return html(rendered_template)


@app.route("/html")
async def rss_html(request):
    url = "http://blog.howie6879.cn/atom.xml"
    feed = parse(url)
    articles = feed['entries']
    data = []
    for article in articles:
        data.append(
            {"title": article["title_detail"]["value"], "link": article["link"], "published": article["published"]})
    return await template('rss.html', articles=articles)
